  def display_page(page)
    prefix = "<span style=\"float: right;\">"
    postfix = "</span>"
    output = prefix
    
    if logged_in?
      if @current_user.has_role?("admin")
        output += edit_page_link_to(page)
      end
    end
    
    output += postfix
    output += page.content
    
    return output
  end
