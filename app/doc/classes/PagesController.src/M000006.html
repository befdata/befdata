  def method_missing(method_id, *args, &block)
    @page = Page.find(:first, :conditions => [ "title = ?", method_id.to_s])
    if @page == nil
      @page = Page.new :content => "<h1>Error 404: Page not found!</h1><p>Page not found!</p>"
    end
    if @page.content == nil
      @page.content = ""
    end
    render :template => "pages/index"
  end
