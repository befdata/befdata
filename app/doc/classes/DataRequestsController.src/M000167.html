  def create
    @data_request = DataRequest.new(params[:data_request])
    author_data_requests = Person.find_all_by_id(params[:people]).
        map{|person| AuthorDataRequest.new(:person => person, :kind => "user")}
    @data_request.author_data_requests = author_data_requests
    @all_persons = Person.all
    unless @data_request.save
      flash[:error] = @data_request.errors.full_messages
      render :action => :new
    else
      redirect_to edit_data_request_path(@data_request)
    end
  end
