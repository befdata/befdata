  def new
    @data_request = DataRequest.new
    @data_request.author = current_user
    @data_request.corresponding = current_user
    @all_persons = Person.all.sort{|a,b| a.to_label <=> b.to_label}

    project = current_user.projects.first
    senior = project.query_by_role(:pi).first
    @data_request.senior_author = senior
    1.times {@data_request.filevalues.build}
  end
