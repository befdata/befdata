  def update
    @person = Person.find_by_id(session[:user_id])

    pwd = params[:person].delete("pwd")
    unless pwd.blank?
      @person.pwd = pwd
    end
    if @person.update_attributes(params[:person])
      flash[:notice] = 'Person was successfully updated.'
      redirect_to :controller => 'profile'
    end
  end
