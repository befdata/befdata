  def load_data_column(co, ch)
    col = Array(@rawdatasheet.row(0)).index(ch) # begin at 0
    measmeth = co.measurements_methodsteps.find_by_columnheader(ch)
    p measmeth.columnheader + " " + col.to_s
    ## load the data
    cldata = Array(@rawdatasheet.column(col))
    cldata.delete_at(0) # first entry is columnheader
    ## return data
    return cldata
  end
