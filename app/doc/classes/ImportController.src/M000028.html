  def check_values(co, ch, cldata, datatype, first, last)
    co = Context.find(co.id)
    measmeth = co.measurements_methodsteps.select{|mm| mm.columnheader == ch}.first
    ms = measmeth.measurements
    (first..last).each do |i|
      n=cldata[i-1]
      p i
      p n
      unless n.nil?
        case datatype
        when "number"
          p ms.select{|m| m.rownr==(i+1)}.first.value.number
        when "cat"
          p ms.select{|m| m.rownr==(i+1)}.first.value.short
          p ms.select{|m| m.rownr==(i+1)}.first.value.long
          p ms.select{|m| m.rownr==(i+1)}.first.value.description
        end
      end
      p "-------------"
    end
  end
