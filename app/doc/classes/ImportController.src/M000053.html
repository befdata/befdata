  def suggest_category_for_entry(portal_cats, sheet_cats, entry)
    cat_found = nil

    # checking categories for exact matches, first on the portal,
    # then in the spreadsheet
    cat = find_entry_in_cat_array(portal_cats, entry)
    logger.debug "------------ cat after portal match  ----------" 
    logger.debug cat.inspect
    unless cat.blank?
      cell_comment = "portal match"
    else       # Is there a match from the sheet? Note in
      # Measurements: sheet match
      cat = find_entry_in_cat_array(sheet_cats, entry) 
      logger.debug "------------ cat after sheet match  ----------" 
      logger.debug cat.inspect
      unless cat.blank?
        # if entry is contained in the sheet, the sheet_cat becomes
        # a portal category for the next data column.  It still has
        # the information in the comment, that it was added from the
        # category sheet (see look_for_provided_cats)
        cell_comment = "sheet match"
      else # in come the "find similar" routines
      end
    end 
    
    if cat.nil?
      cell_comment = "created during upload"
      cat = Categoricvalue.
        create(:short => entry, :long => entry, :description => entry,
               :comment => "created during upload")
    end
    
    return {:cell_comment => cell_comment,
      :cat => cat}
  end
