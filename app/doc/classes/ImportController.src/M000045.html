  def numeric_data_column_import(data_column_id)
    data_column = MeasurementsMethodstep.find(data_column_id)

    # sheet_cats = data_column.import_categories; they don't change
    sheet_cats = data_column.import_categories.map{|icat| icat.categoricvalue}
    
    # the entries themselves
    cells = data_column.measurements

    # Check each entry loop
    cells.each do |cell| 
      entry = cell.import_value
      obs = cell.observation

      # here are the portal cats; array of category instances; they
      # change, if a new category is linked to a cell
      portal_cats = data_column.methodstep.datacell_categories

      # a new numeric value immediately converts anything to a float,
      # so we have to check here
      if numeric?(entry)
        # entry is convertable to a float
        value = Numericvalue.create(:number => entry)
        comment = "valid"
      else # create categoric value
        comment_cat_hash = 
          suggest_category_for_entry(portal_cats, sheet_cats,
                                     entry)
        value = comment_cat_hash[:cat]
        comment = comment_cat_hash[:cell_comment]
      end # asking for numericality

      cell.value = value
      cell.comment = comment
      cell.save
    end # Entry loop
  end
