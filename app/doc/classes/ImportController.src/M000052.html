   def cell_category_update
     if logged_in?
       first_cell = Measurement.find(params[:measurement][:id])
       same_entry_cells = first_cell.same_entry_cells

       # category
       txt1 = "manually approved"
       cat = first_cell.categoricvalue
       cat.update_attributes(:comment => txt1)
       
       # updating each cell
       txt2 = "valid"
       same_entry_cells.each do |cell| 
         old_cat = cell.categoricvalue
         cell.comment = txt2
         cell.update_attributes(params[:measurement])
         old_cat.destroy
       end 
       
       # !! no validation yet!!
       redirect_to :back
       
     else
       # Not logged in, redirect to login form
       session[:return_to] = request.request_uri
       redirect_to login_path and return
     end
   end
