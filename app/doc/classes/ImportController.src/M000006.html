  def freeformat_file_upload
    if logged_in?
      filevalue = Filevalue.new(params[:filevalue])

      begin 
        Filevalue.transaction do
          logger.debug "---- In freeformat_file_upload"
          filevalue.save

          # Creating the context
          filename = filevalue.file_file_name
          title = "Free format file " + filevalue.file_file_name
          abstract = "Dummy text for importing a free format file, please replace"
          filename_noending = filename.split('.')[0]
          export_filename = filename_noending + ".xls" # this is the ouput filename
          @context = Context.create(:title => title,
                                    :abstract => abstract,
                                    :filename => export_filename)

          # Creating the owner
          pers_role = @current_user.person_roles[0]
          logger.debug "pers_role.inspect"
          logger.debug pers_role.inspect
          @context.person_roles << pers_role
          logger.debug "@context.person_roles"
          logger.debug @context.person_roles.inspect

          # Creating the data column
          dc = MeasurementsMethodstep.new(:columnheader => "file",
                                          :definition => "free format file",
                                          :columnnr => 1)
          logger.debug "dc.inspect"
          logger.debug dc.inspect
          # Creating a new Data group by default
          dg = Methodstep.create(:title => title, :description => title)
          logger.debug "dg.inspect"
          logger.debug dg.inspect
          dc.methodstep = dg
          dc.context = @context
          dc.save
          logger.debug "dc.inspect"
          logger.debug dc.inspect

          # Adding file to sheet cell
          obs = Observation.create(:rownr => 2)
          cl = Measurement.create(:observation => obs,
                                  :value => filevalue,
                                  :measurements_methodstep => dc)

          redirect_to(:controller => :import, :action => :freeformat_context_edit, 
                      :context_id => @context.id)
        end
      rescue ActiveRecord::RecordInvalid => invalid
        redirect_to :root
      end
    end
  end
