  def find_new_categories_to_be_added(raw_cldata, datalookup)
    ## key begins with 0, ruby type locator in cldata
    newdatalookup_array = datalookup.select{|k,v| v.nil?}
    ## find unique categories from the raw data without known categories
    unique_new_categories = newdatalookup_array.collect{|i| raw_cldata[i[0]]}.uniq
    p "These new categories will be added to the known categories:"
    p unique_new_categories
    return(unique_new_categories)
  end
