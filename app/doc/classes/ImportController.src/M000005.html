  def create_dataset_filevalue
      filevalue = Filevalue.new(params[:filevalue])

      begin 
        Filevalue.transaction do
          filevalue.save
          #! should go the context upload action
          redirect_to(:controller => :contexts, :action => :upload, 
                      :filevalue_id => filevalue.id)
        end
      rescue ActiveRecord::RecordInvalid => invalid
        redirect_to :back
        #! showing the message that file upload did not work
      end
  end
