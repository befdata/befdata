  def category_data_column_import(data_column_id)
    data_column = MeasurementsMethodstep.find(data_column_id)
    
    # here are the portal cats; array of category instances
    portal_cats = data_column.methodstep.datacell_categories

    # sheet_cats = data_column.import_categories
    sheet_cats = data_column.import_categories.
      map{|icat| icat.categoricvalue}

    # the entries themselves
    cells = data_column.measurements

    # Check each entry loop
    cells.each do |cell| 
      entry = cell.import_value
      obs = cell.observation
      
      # here are the portal cats; array of category instances; they
      # change, if a new category is linked to a cell
      portal_cats = data_column.methodstep.datacell_categories

      comment_cat_hash = 
        suggest_category_for_entry(portal_cats, sheet_cats,
                                   entry)
      cat = comment_cat_hash[:cat]
      comment = comment_cat_hash[:cell_comment]
      comment = "valid" if comment = "portal match"
      
      cell.comment = comment
      cell.value = cat
      cell.save
    end # Entry loop

  end
