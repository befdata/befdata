  def submethod_summary(submethod)
    submethod_mms = submethod.measurements
    if submethod_mms.length > 0
      submDataType = submethod_mms[0].value_type
    else
      submDataType = "No Measurements provided for this column yet"
    end

    # get the meaningful values
    if submDataType == "Categoricvalue"
      submData =  submethod.measurements.collect{|m| m.categoricvalue.long}
    end
    if submDataType == "Numericvalue"
      submData =  submethod.measurements.collect{|m| m.numericvalue.number}
    end
    submData = submData.compact.uniq

    @submethodSummary = {:submeth => submethod,
      :datatype => submDataType, :data => submData
    }
  end
