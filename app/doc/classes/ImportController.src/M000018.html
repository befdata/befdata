  def update_import_category_new_category 
    if logged_in?
      icat = ImportCategory.find(params[:import_category][:id])
      oldcat = icat.categoricvalue
      newcat = Categoricvalue.new(params[:categoric_value])
      icat.approved = true
      dh = icat.measurements_methodstep
      val = icat.raw_data_value
      # data cells with this raw data value have to get the new
      # category
      dcs = dh.measurements.select{|dc| dc.import_value == val}

      logger.debug "---  creating new category for an existing icat -----"
      logger.debug icat.inspect
      logger.debug params.inspect
      logger.debug dh


      begin 
        ImportCategory.transaction do
          newcat.save
          icat.categoricvalue = newcat
          icat.save
          # add the new value to the cell values
          dcs.each do |dc|
            dc.value = newcat
            dc.save
          end
          # get rid of unassociated categories; will not destroy
          # categories linked to other cells or import_categories
          oldcat.destroy
          redirect_to :back
        end
      rescue ActiveRecord::RecordInvalid => invalid
        redirect_to :root
      end
    else
      # Not logged in, redirect to login form
      session[:return_to] = request.request_uri
      redirect_to login_path and return
    end
  end
