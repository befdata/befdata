  def update_data_group
    if logged_in?
      data_header = MeasurementsMethodstep.find(params[:measurements_methodstep][:id])
      data_group = Methodstep.new(params[:methodstep])

      logger.debug "--------------------------------------------------"
      logger.debug data_header.inspect
      logger.debug data_group.inspect
      logger.debug params.inspect

      begin 
        MeasurementsMethodstep.transaction do
          if data_group.save
            data_header.methodstep = data_group
            data_header.save
            redirect_to :back
          end
        end
      rescue ActiveRecord::RecordInvalid => invalid
        redirect_to :root
      end
    else
      # Not logged in, redirect to login form
      session[:return_to] = request.request_uri
      redirect_to login_path and return
    end
  end
