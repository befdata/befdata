  def destroy
    if logged_in?

      @filevalue = Filevalue.find(params[:id])
      @filevalue.destroy
      
      respond_to do |format|
        format.html { redirect_to(filevalues_url) }
        format.xml  { head :ok }
      end
    else
      # Not logged in, redirect to login form
      session[:return_to] = request.request_uri
      redirect_to login_path and return
    end

  end
