  def create_cart_data_column
    @data_column = MeasurementsMethodstep.find(params[:measurements_methodstep_id])
    @cart_data_column = CartDataColumn.create!(:cart => current_cart, :measurements_methodstep => @data_column)
    flash[:notice] = "Added #{@data_column.columnheader} to cart."
    redirect_to :back
  end
