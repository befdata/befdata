  def try_retrieving_projects_from_tag_list(book)
    return if book.tag_list.blank?
    book.tag_list.split(",").each do |t|
      Project.find_by_converting_to_tag(t).each do |p|
        self.projects << p unless self.projects.include? p
      end
    end
  end
