  def abstract_with_freeformats
    f_strings = self.freeformats.collect do |f|
      "File asset " + f.file_file_name + (f.description.blank? ? "" : (": " + f.description))
    end
    self.abstract + (f_strings.empty? ? "" : (" - " + f_strings.join(" - ")))
  end
