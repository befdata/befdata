  def eml_unit_for_column(column)
    case column.unit
      when 'meter' || 'm'
        "meter"
      when 'millimeter' || 'mm'
        "millimeter"
      when 'gram' || 'g'
        "gram"
      when 'gramsPerSquareMeter' || 'g/m^2'
      else
        "dimensionless"
    end
  end
