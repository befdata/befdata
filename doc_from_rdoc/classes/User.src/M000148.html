  def projectroles
    # return the project roles that this user has
    # disclude any admins roles
    @projectsarray = []
    @index =0
    for role in self.role_objects.reject{|r| r.name == "admin"}
      if(role.authorizable.class.to_s == "Project")
        @projectsarray[@index]= role
        @index = @index +1
      end
    end
    @projectsarray
  end
