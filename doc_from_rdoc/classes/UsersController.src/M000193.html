  def show
    redirect_to(:action => "index") and return if params[:id].blank?

    u_id = params[:id].split(/-/).first
    @user = User.find u_id

    @user_datasets_owned = @user.datasets_owned.sort_by {|d| d.title.to_s}

    redirect_to(:action => "index", :status => :not_found) unless @user
  end
