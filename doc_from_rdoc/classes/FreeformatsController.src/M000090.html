  def create
    freeformat = @freeformattable.freeformats.build(params[:freeformat])
    if freeformat.save
      redirect_to :back
    else
      flash[:error] = "#{freeformat.errors.to_a.first.capitalize}"
      redirect_to :back
    end
  end
