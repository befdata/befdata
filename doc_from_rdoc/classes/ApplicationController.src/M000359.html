  def access_denied
    if current_user
      flash[:error] = 'Access denied. You do not have the appropriate rights to perform this operation.'
      redirect_back_or_default root_url
    else
      flash[:error] = 'Access denied. Try to log in first.'
      redirect_back_or_default root_url
    end
  end
