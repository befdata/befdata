  def portal_users_listed_as_responsible
    portal_users = []
    if @book.nil?
      load_datafile
    end
    members_listed_as_responsible.each do |member|
      portal_users << User.find_by_firstname_and_lastname(clean_string(member[0]), clean_string(member[1]))
    end
    portal_users.compact
  end
